// Backup do código corrigido function renderizarDetalhesOracle(data) { const cliente = data.cliente; const pedidos = data.pedidos_oracle || []; const itens = data.itens_oracle || []; let html = `
Dados do Cliente
${cliente.cd_cliente_oracle ? ` ` : ''} ${cliente.representante_oracle ? ` ` : ''}
Nome: ${cliente.nome}
CNPJ: ${cliente.cnpj || '-'}
Telefone: ${cliente.telefone ? ` ${cliente.telefone} ` : '-' }
Código Oracle: ${cliente.cd_cliente_oracle}
Representante Oracle: ${cliente.representante_oracle}
Dados de Crédito
${cliente.categoria_consultor ? ` ` : ''} ${cliente.ultimo_pedido_oracle ? ` ` : ''} ${cliente.valor_ultimo_pedido ? ` ` : ''} ${cliente.situacao_ultimo_pedido ? ` ` : ''}
Conceito: ${cliente.conceito || 'SEM CONCEITO'}
Consultor Oracle: ${cliente.categoria_consultor}
Último Pedido: ${cliente.ultimo_pedido_oracle}
Valor Último Pedido: R$ ${parseFloat(cliente.valor_ultimo_pedido).toFixed(2)}
Situação: ${cliente.situacao_ultimo_pedido === 'P' ? 'Pago' : cliente.situacao_ultimo_pedido === 'C' ? 'Cancelado' : cliente.situacao_ultimo_pedido}
`; // Adicionar itens dos pedidos se houver if (itens.length > 0) { // Deduplicação simples const itensUnicos = []; const vistos = new Set(); itens.forEach(item => { const chave = `${item.idproduto}_${item.quantidade}_${item.precounitario}`; if (!vistos.has(chave)) { vistos.add(chave); itensUnicos.push(item); } }); // Agrupar por pedido const itensPorPedido = {}; itensUnicos.forEach(item => { const pedido = item.cd_pedido || 'sem-pedido'; if (!itensPorPedido[pedido]) { itensPorPedido[pedido] = { dt_pedido: item.dt_pedido, cd_pedido: item.cd_pedido, itens: [] }; } itensPorPedido[pedido].itens.push(item); }); html += `
Itens dos Pedidos (últimos 365 dias) ${Object.keys(itensPorPedido).length} pedidos
`; // Renderizar cada pedido Object.values(itensPorPedido).forEach(pedido => { const data = pedido.dt_pedido ? new Date(pedido.dt_pedido).toLocaleDateString('pt-BR') : '-'; const numeroPedido = pedido.cd_pedido || '-'; html += `
Pedido #${numeroPedido} - ${data}
`; pedido.itens.forEach(item => { const nomeProduto = item.nome_produto || item.idproduto || '-'; const quantidade = item.quantidade || 0; const preco = item.precounitario ? `R$ ${parseFloat(item.precounitario).toFixed(2)}` : '-'; html += `
${nomeProduto}
${quantidade} unid - ${preco}
`; }); html += `
`; }); html += `
`; } // Adicionar informações de sincronização if (cliente.data_ultima_sincronizacao) { html += `
Informações: Dados sincronizados com Oracle em ${cliente.data_ultima_sincronizacao}
`; } // Atualizar o conteúdo do modal document.getElementById('detalhesOracleContent').innerHTML = html; // Guardar ID do cliente para usar no botão de ligação window.clienteDetalhesAtual = cliente.id; }
